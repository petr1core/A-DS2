cmake_minimum_required(VERSION 3.5)
set(PROJECT_NAME Expression)
project(${PROJECT_NAME})

set(CMAKE_CONFIGURATION_TYPES "Debug;Release" CACHE STRING "Configs" FORCE)
if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()

include_directories(${CMAKE_CURRENT_SOURCE_DIR}/ExpressionLib)
add_subdirectory(ExpressionLib)
add_subdirectory(ExpressionTest)

file(GLOB srcs "*.cpp")
add_executable(${PROJECT_NAME} ${srcs})
target_link_libraries(${PROJECT_NAME} ${ExpressionLib} gtest)
