cmake_minimum_required(VERSION 3.5)
set(PROJECT_NAME HierarchyList)
project(${PROJECT_NAME} CXX)

set(CMAKE_CONFIGURATION_TYPES "Debug;Release" CACHE STRING "Configs" FORCE)
if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()

include_directories(${CMAKE_CURRENT_SOURCE_DIR}/HierarchyListLib)
add_subdirectory(HierarchyListLib)
add_subdirectory(HierarchyListTest)

file(GLOB srcs "*.cpp")
add_executable(${PROJECT_NAME} ${srcs})
target_link_libraries(${PROJECT_NAME} ${HierarchyListLib} gtest)
